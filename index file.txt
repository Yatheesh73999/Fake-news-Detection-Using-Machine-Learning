    <script>
        // ── HELPERS ──────────────────────────────────────────────────────────────────
        function $(id) { return document.getElementById(id) }

        function fillClaim(text) {
            $('claimInput').value = text;
            $('claimInput').focus();
        }

        function hideAll() {
            ['loader', 'errorBox', 'verdictWrap', 'evidenceSection'].forEach(id => {
                $(id).classList.remove('show');
            });
        }

        function showError(msg) {
            hideAll();
            $('errorDetail').textContent = msg;
            $('errorBox').classList.add('show');
            $('verifyBtn').disabled = false;
        }

        // ── PIPELINE ANIMATION ───────────────────────────────────────────────────────
        const STEP_MSGS = [
            'Fetching related news articles...',
            'Evaluating source credibility & relevance...',
            'Detecting stance & semantic similarity...',
            'Scoring evidence strength...',
            'Generating final verdict...',
        ];

        let pipeTimer = null;

        function resetPipeline() {
            for (let i = 1; i <= 5; i++) $(('pn' + i)).className = 'pipe-node';
            for (let i = 1; i <= 4; i++) $(('pc' + i)).className = 'pipe-connector';
        }

        function animatePipeline() {
            resetPipeline();
            let step = 0;
            function tick() {
                if (step > 0) {
                    $('pn' + step).className = 'pipe-node done';
                    if (step <= 4) $('pc' + step).className = 'pipe-connector lit';
                }
                if (step < 5) {
                    $('pn' + (step + 1)).className = 'pipe-node active';
                    $('loaderStatus').innerHTML = STEP_MSGS[step] + '<span>...</span>';
                }
                step++;
                if (step <= 5) pipeTimer = setTimeout(tick, 1000 + Math.random() * 400);
            }
            tick();
        }

        function finishPipeline() {
            clearTimeout(pipeTimer);
            for (let i = 1; i <= 5; i++) $('pn' + i).className = 'pipe-node done';
            for (let i = 1; i <= 4; i++) $('pc' + i).className = 'pipe-connector lit';
        }

        // ── VERDICT RENDER ───────────────────────────────────────────────────────────
        function verdictMeta(v) {
            if (v === 'LIKELY_TRUE') return { emoji: '✅', label: 'LIKELY TRUE' };
            if (v === 'LIKELY_FALSE') return { emoji: '❌', label: 'LIKELY FALSE' };
            return { emoji: '❓', label: 'UNVERIFIED' };
        }

        function renderVerdict(data) {
            const verdict = data.verdict || {};
            const raw = (verdict.verdict || 'UNVERIFIED').toUpperCase().replace(/ /g, '_');
            const conf = parseFloat(verdict.confidence || 0);
            const reason = verdict.reason || 'Insufficient evidence to determine.';
            const evidences = data.evidences || [];
            const supports = evidences.filter(e => (e.stance || '').toUpperCase() === 'SUPPORTS').length;
            const refutes = evidences.filter(e => (e.stance || '').toUpperCase() === 'REFUTES').length;
            const meta = verdictMeta(raw);

            const card = $('verdictCard');
            card.setAttribute('data-v', raw);
            $('vEmoji').textContent = meta.emoji;
            $('vHeadline').textContent = meta.label;
            $('vReason').textContent = reason;
            $('vConf').textContent = (conf * 100).toFixed(0) + '%';
            $('vArticles').textContent = data.articles_used ?? evidences.length;
            $('vSupports').textContent = supports;
            $('vRefutes').textContent = refutes;

            $('verdictWrap').classList.add('show');
            setTimeout(() => { $('confFill').style.width = (conf * 100) + '%'; }, 80);
        }

        // ── EVIDENCE RENDER ──────────────────────────────────────────────────────────
        function safeDate(raw) {
            if (!raw) return 'N/A';
            try { return new Date(raw).toLocaleDateString('en-IN', { day: 'numeric', month: 'short', year: 'numeric' }); }
            catch { return raw; }
        }

        function renderEvidence(evidences) {
            const grid = $('evGrid');
            grid.innerHTML = '';

            if (!evidences || evidences.length === 0) {
                grid.innerHTML = '<p style="font-family:\'IBM Plex Mono\',monospace;font-size:12px;color:var(--muted);padding:24px 0">No evidence articles returned from pipeline.</p>';
                $('evidenceSection').classList.add('show');
                return;
            }

            $('evMetaCount').textContent = evidences.length + ' SOURCES ANALYZED';

            evidences.forEach((ev) => {
                const stance = (ev.stance || 'NEUTRAL').toUpperCase();
                const cred = (ev.credibility || 'LOW').toUpperCase();
                const strength = (ev.evidence_strength || 'WEAK').toUpperCase();
                const sim = parseFloat(ev.similarity || 0);
                const score = parseFloat(ev.score || 0);
                const sentence = ev.best_sentence || 'Content not available.';
                const stanceReason = ev.stance_reason || '';
                const url = ev.url || '#';
                const title = ev.title || 'Untitled Article';
                const source = ev.source || 'Unknown Source';
                const date = safeDate(ev.published_date || ev.date);
                const direct = ev.direct_match;

                const card = document.createElement('div');
                card.className = 'ev-card';
                card.setAttribute('data-stance', stance);

                card.innerHTML = `
      <div class="ev-top">
        <div class="stance-tag">${stance}</div>
        <div class="ev-meta-col">
          <div class="ev-art-title" title="${escHtml(title)}">${escHtml(title)}</div>
          <div class="ev-source-row">
            <span class="ev-source">${escHtml(source)}</span>
            <span class="ev-date">${escHtml(date)}</span>
            <span class="cred-tag ${cred}">${cred}</span>
            ${direct ? '<span class="direct-tag">✓ Direct Match</span>' : ''}
          </div>
        </div>
        <button class="ev-toggle" onclick="toggleCard(this)">DETAILS ▾</button>
      </div>
      <div class="ev-body">
        <div class="ev-sentence">"${escHtml(sentence)}"</div>
        ${stanceReason ? `<div class="ev-reason-text">Stance reason: ${escHtml(stanceReason)}</div>` : ''}
        <div class="scores-row">
          <div class="sc-item"><div class="sc-key">Similarity</div><div class="sc-val">${(sim * 100).toFixed(0)}%</div></div>
          <div class="sc-item"><div class="sc-key">Evidence Score</div><div class="sc-val">${(score * 100).toFixed(0)}%</div></div>
        </div>
        <div class="strength-pill ${strength}">${strength}</div>
        ${url && url !== '#' && url !== 'N/A'
                        ? `<br><a class="ev-link" href="${url}" target="_blank" rel="noopener">↗ READ FULL ARTICLE</a>`
                        : ''}
      </div>`;

                grid.appendChild(card);
            });

            $('evidenceSection').classList.add('show');
        }

        function toggleCard(btn) {
            const body = btn.closest('.ev-card').querySelector('.ev-body');
            const open = body.classList.toggle('open');
            btn.textContent = open ? 'CLOSE ▴' : 'DETAILS ▾';
        }

        function escHtml(str) {
            return String(str)
                .replace(/&/g, '&amp;')
                .replace(/</g, '&lt;')
                .replace(/>/g, '&gt;')
                .replace(/"/g, '&quot;');
        }

        // ── MAIN FLOW ─────────────────────────────────────────────────────────────────
        async function runVerification() {
            const claim = $('claimInput').value.trim();
            if (!claim) { alert('Please enter a claim to verify.'); return; }

            hideAll();
            clearTimeout(pipeTimer);
            $('verifyBtn').disabled = true;
            $('loader').classList.add('show');
            animatePipeline();
  
            try {
                const API_URL = "http://127.0.0.1:5000/verify";
            const resp = await fetch(API_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ claim })
                });

                if (!resp.ok) {
                    const err = await resp.json().catch(() => ({}));
                    throw new Error(err.error || 'Server returned ' + resp.status);
                }

                const data = await resp.json();
                console.log("DATA FROM BACKEND:", data);

                // Let animation finish gracefully
                await new Promise(r => setTimeout(r, 500));
                finishPipeline();
                await new Promise(r => setTimeout(r, 350));
                $('loader').classList.remove('show');

                renderVerdict(data);
                renderEvidence(data.evidences || []);

            } catch (err) {
                clearTimeout(pipeTimer);
                let msg = err.message || 'Unknown error occurred.';
                if (msg.toLowerCase().includes('fetch') || msg.toLowerCase().includes('failed')) {
                    msg = 'Cannot reach backend at 127.0.0.1:5000.\n\nMake sure api.py is running:\n\n  python api.py';
                }
                showError(msg);
            } finally {
                $('verifyBtn').disabled = false;
            }
        }

        // Enter to submit
        $('claimInput').addEventListener('keydown', e => {
            if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); runVerification(); }
        });
